import svgPanZoom from "svg-pan-zoom"; import { stream } from "m2" export default ({ targets: [{ node: svg }], source }) => { svg.setAttribute("width", window.screen.width); svg.setAttribute("height", window.screen.height); return stream( (emt, { sweep, over }) => { let instance = null; const observer = new MutationObserver( () => { if(!instance && document.contains(svg)) { instance = svgPanZoom(svg, { refreshRate: "auto", viewportSelector: "g", zoomScaleSensitivity: 0.3, fit: 1 }); } } ); observer.observe( document, { childList: true, subtree: true } ); sweep.add( () => { if(instance) { observer.disconnect(); instance.destroy(); instance = null; } } ); over.add(source.on( emt )); } ); }
